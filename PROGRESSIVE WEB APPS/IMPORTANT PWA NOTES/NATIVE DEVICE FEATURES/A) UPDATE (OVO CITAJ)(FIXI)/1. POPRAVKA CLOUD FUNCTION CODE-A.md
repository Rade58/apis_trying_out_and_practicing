                                // IMAGE URL MORAM FORMIRATI OD 'PARCIJALNIH' INFORMACIJA

                        )

                        // ODAVDE BI ISAO CODE KOJI JE REZERVISAN ZA PUSH (PODESAVANJE VAPID KLJUCEVA ITD)
